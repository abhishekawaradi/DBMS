DECLARE
    CURSOR c1 IS 
        SELECT * FROM O_RollCall;

    roll_no NUMBER;
    counter NUMBER := 0;
BEGIN
    FOR c1_rec IN c1 LOOP
        SELECT COUNT(*) INTO roll_no 
        FROM N_RollCall 
        WHERE roll = c1_rec.roll;
        IF roll_no = 0 THEN
            INSERT INTO N_RollCall 
            VALUES (c1_rec.roll, c1_rec.name);
            counter := counter + 1;
        END IF;
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('Number of Rows Inserted: ' || counter);
END;
