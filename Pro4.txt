CREATE TABLE CUSTOMERS ( 
    ID NUMBER(5) PRIMARY KEY, 
    NAME VARCHAR2(50), 
    AGE NUMBER(3), 
    ADDRESS VARCHAR2(100), 
    SALARY NUMBER(10,2) 
);

CREATE OR REPLACE TRIGGER salary_diff_trigger 
BEFORE INSERT OR UPDATE OR DELETE 
ON CUSTOMERS 
FOR EACH ROW 
DECLARE 
    salary_difference NUMBER; 
BEGIN 
    IF UPDATING THEN 
        salary_difference := :NEW.SALARY - :OLD.SALARY; 
        DBMS_OUTPUT.PUT_LINE('Salary Difference: ' || salary_difference); 
    ELSIF INSERTING THEN 
        DBMS_OUTPUT.PUT_LINE('New Salary Inserted: ' || :NEW.SALARY); 
    ELSIF DELETING THEN 
        DBMS_OUTPUT.PUT_LINE('Deleted Salary Was: ' || :OLD.SALARY); 
    END IF; 
END; 
/ 
SET SERVEROUTPUT ON; 
UPDATE CUSTOMERS 
SET SALARY = 35000 
WHERE ID = 1;
