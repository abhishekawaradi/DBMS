    E_id NUMBER(5) PRIMARY KEY, 
    E_name VARCHAR2(50), 
    Age NUMBER(3), 
    Salary NUMBER(10,2)

SELECT * FROM Employee; 
SELECT COUNT(E_name) AS Total_Employees 
FROM Employee; 
SELECT MAX(Age) AS Maximum_Age 
FROM Employee; 
SELECT MIN(Age) AS Minimum_Age 
FROM Employee; 
SELECT E_name, Salary 
FROM Employee 
ORDER BY Salary ASC; 
SELECT Salary, COUNT(*) AS No_of_Employees 
FROM Employee 
GROUP BY Salary 
ORDER BY Salary;
